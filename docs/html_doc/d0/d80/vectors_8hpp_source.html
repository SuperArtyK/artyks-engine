160 inline bool comparevec(const A& _vec1, const B& _vec2, const int comptype, const float _epsilon = std::numeric_limits<float>::epsilon()) {
161  static_assert(
162  (std::is_same<A, vec2>::value || std::is_same<A, vec2int>::value) ||
163  (std::is_same<B, vec2>::value || std::is_same<B, vec2int>::value) ||
164  (std::is_same<A, vec3>::value || std::is_same<A, vec3int>::value) ||
165  (std::is_same<B, vec3>::value || std::is_same<B, vec3int>::value)
166  , "[engine]: comparevec: You cannot use non-vector here.");
167 
168 
169  const float diff1 = _vec1.x - _vec2.x;
170  const float diff2 = _vec1.y - _vec2.y;
171  float diff3;
172  if constexpr ((std::is_same<A, vec3>::value || std::is_same<A, vec3int>::value) && (std::is_same<B, vec3>::value || std::is_same<B, vec3int>::value)) {
173  diff3 = _vec1.z - _vec2.z;
174  }
175  else {
176  diff3 = diff2;
177  }
178  switch (comptype)
179  {
180  //equality
181  case 0:
182  return (diff1 < _epsilon && -diff1 < _epsilon) && (diff2 < _epsilon && -diff2 < _epsilon) && (diff3 < _epsilon && -diff3 < _epsilon);
183  break;
184 
185  //not-equality
186  case 1:
187  return (diff1 > _epsilon || -diff1 > _epsilon) || (diff2 > _epsilon || -diff2 > _epsilon) || (diff3 > _epsilon || -diff3 > _epsilon);
188  break;
189 
190  //less than
191  case 2:
192  return (diff1 < -_epsilon) || (diff2 < -_epsilon) || (diff3 < -_epsilon);
193  break;
194 
195  //more than
196  case 3:
197  return (diff1 > _epsilon) || (diff2 > _epsilon) || (diff3 > _epsilon);
198  break;
199 
200  //less than or equals to
201  case 4:
202  return (diff1 <= _epsilon) && (diff2 <= _epsilon) && (diff3 <= _epsilon);
203  break;
204 
205  //more than

208  break;
