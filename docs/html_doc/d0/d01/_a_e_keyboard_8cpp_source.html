240 void AEKeyboard::mainthread_keys(void) {
241  //keyboard
242  AEFrame m_fr(GAME_FPS*2);
243  DWORD m_event_amt = 0;
244  INPUT_RECORD m_event_buffer[32];//32 is just enough
245  artyk::utils::debug_log(m_logptr, "Started keyscanning sequence.", LOG_OK, m_modulename);
246 
247 
248  artyk::utils::waitfortick();
249  while (!AEKeyboard::__doneinit) {
250  m_fr.sleep();
251  }
252  m_fr.setfps(GAME_FPS);
253  while (!m_bstoptrd)
254  {//for stopping the thread
255 #ifdef AE_KB_DELAY
256  m_fr.sleep();//sleep frame if we are not in focus
257 #endif // AE_KB_DELAY
258 
259  while (artyk::g_console_hwnd == GetForegroundWindow())
260  {
261  //get keyboard state
262  for (int i = 0; i < 256; i++)
263  {
264 
265  m_keys[i].m_state = GetAsyncKeyState(i);
266  {
267  if (m_keys[i].m_state != 0) {
268  if (m_keyOld[i] == 0) {
269  m_keys[i].m_isPressed = true;
270  m_keys[i].m_isHeld = false;
271  m_keys[i].m_isReleased = false;
272  m_keys[i].m_isUsed = true;
273 
274  }
275  else
276  {
277  m_keys[i].m_isPressed = false;
278  m_keys[i].m_isHeld = true;
279  m_keys[i].m_isReleased = false;
280  m_keys[i].m_isUsed = true;
281  }
282  }
283  else {
284  if (m_keyOld[i] != 0) {
285  m_keys[i].m_isPressed = false;
286  m_keys[i].m_isHeld = false;
287  m_keys[i].m_isReleased = true;
288  m_keys[i].m_isUsed = true;
289  }
290  else {
291  m_keys[i].m_isPressed = false;
292  m_keys[i].m_isHeld = false;
293  m_keys[i].m_isReleased = false;
294  m_keys[i].m_isUsed = false;
295  }
296  }
297  }
298  //if (i == VK_SPACE) {
299  // std::cout << m_keys[i].m_state << " " << m_keyOld[i] << NLC;
300  //}
301  m_keyOld[i] = m_keys[i].m_state;
302  }
303  if (m_enablemouse) {//only if we want support for mouse
304  //get mouse events
305 
306  GetNumberOfConsoleInputEvents(artyk::g_input_handle, &m_event_amt);
307 
308  if (m_event_amt > 0) {
309  ReadConsoleInput(
310  artyk::g_input_handle, m_event_buffer,
311  (m_event_amt>
312  (arrsize(m_event_buffer))//doing this only for safety, if the input amt surpasses the size of the buffer
313  )
314  ? (arrsize(m_event_buffer))
315  : m_event_amt, &m_event_amt);
316  //we pass the even amount as for "lpNumberOfEventsRead" for 2 things: 1 saves space, we don't allocate temp stuff; 2 we can see how many we read.
317  }
318 
319  //check for handle events
320  for (DWORD i = 0; i < m_event_amt; i++)
321  {
322  if (m_event_buffer[i].EventType == MOUSE_EVENT && m_event_buffer[i].Event.MouseEvent.dwEventFlags == MOUSE_MOVED) {
323 
324  m_mousepos[0] = m_event_buffer[i].Event.MouseEvent.dwMousePosition.X;
325  m_mousepos[1] = m_event_buffer[i].Event.MouseEvent.dwMousePosition.Y;
326 
327  }
328  }
329 
330  }
331 
332 
333  if (m_bstoptrd) {
334  artyk::utils::debug_log(m_logptr, "Stopped keyscanning sequence.", LOG_OK, m_modulename);
335  return;
336  }
337  else
338  {
339 #ifdef AE_KB_DELAY
340  m_fr.sleep();
341 #endif // AE_KB_DELAY
342  }
343  }
344 
345 
346  }
