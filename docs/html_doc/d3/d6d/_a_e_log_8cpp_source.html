102  int CPUInfo[4] = { -1 };
103  uint nExIds;
104  char CPUBrandString[0x40];
105  // Get the information associated with each extended ID.
106  __cpuid(CPUInfo, 0x80000000);
107  nExIds = CPUInfo[0];
108  for (uint i = 0x80000000; i <= nExIds; ++i)
109  {
110  __cpuid(CPUInfo, i);
111  // Interpret CPU brand string
112  if (i == 0x80000002)
113  memcpy(CPUBrandString, CPUInfo, sizeof(CPUInfo));
114  else if (i == 0x80000003)
115  memcpy(CPUBrandString + 16, CPUInfo, sizeof(CPUInfo));
116  else if (i == 0x80000004)
117  memcpy(CPUBrandString + 32, CPUInfo, sizeof(CPUInfo));
118  }
119  //string includes manufacturer, model and clockspeed
120  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: CPU Name: " + CPUBrandString + "\n");

123  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Processor type: " + to_string(sysinf.dwProcessorType) + "\n");
124  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Active processor mask: " + to_string(sysinf.dwActiveProcessorMask) + "\n");
125  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Memory Page Size: " + to_string(sysinf.dwPageSize) + "\n");
126  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Memory allocation granularity: " + to_string(sysinf.dwAllocationGranularity) + "\n");
127  MEMORYSTATUSEX statex;

129  GlobalMemoryStatusEx(&statex);
130  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Total System Memory: " + to_string(statex.ullTotalPhys / 1024) + "KB\n");
131  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Free System Memory: " + to_string(statex.ullAvailPhys / 1024) + "KB\n");

134  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Free Page File: " + to_string(statex.ullAvailPageFile / 1024) + "KB\n");
135  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Total Process Memory: " + to_string(statex.ullTotalVirtual / 1024) + "KB\n");
136  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Free Process Memory: " + to_string(statex.ullAvailVirtual / 1024) + "KB\n");
137  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Minimum Application Address: " + artyk::utils::addrtostr(sysinf.lpMinimumApplicationAddress) + "\n");
138  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: Maximum Application Address: " + artyk::utils::addrtostr(sysinf.lpMaximumApplicationAddress) + "\n");
139  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: OEM ID: " + to_string(sysinf.dwOemId) + "\n");
140  m_message.append("[" + artyk::utils::currentDateTime() + "][" + checktype(LOG_INFO) + "][Engine]: ----------END SYSTEM INFO----------\n");
141 #endif // AE_LOG_SYSTEM_INFO
142  engstarted.write(m_message.c_str(), m_message.size());

144  m_entrycount++;
145  }
