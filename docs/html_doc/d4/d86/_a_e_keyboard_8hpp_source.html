155  std::memset(AEKeyboard::m_keyOld, 0, 256 * sizeof(short));
156  std::memset(AEKeyboard::m_keys, 0, 256 * sizeof(AEKBKey));
157 
158  for (short i = 0; i < 256; i++) {
159  AEKeyboard::m_keys[i].m_keyid = i;
160  for (int a = 0; a < arrsize(AEKeyboard::m_keycodes); a++) {
161  if (i == AEKeyboard::m_keycodes[a].second) {
162  AEKeyboard::m_keys[i].m_name = AEKeyboard::m_keycodes[a].first;
163  }
164  }
165  if (AEKeyboard::m_keys[i].m_name == nullptr) {
166  AEKeyboard::m_keys[i].m_name = "---";
167  }
168  }
169  //done initializing
170  AEKeyboard::__doneinit = true;
171  return 0;
172 }
