276 void AEGraphic::drawLine(const vec2int& myvec2_1, const vec2int& myvec2_2, const CHAR_INFO& mych) {
277 
278  if (myvec2_1.x == myvec2_2.x && myvec2_1.y == myvec2_2.y) {
279  setPixel(myvec2_1, mych);
280  return;
281  }
282 
283  const int
284  deltaX = abs(myvec2_2.x - myvec2_1.x),
285  deltaY = abs(myvec2_2.y - myvec2_1.y),
286  signX = myvec2_1.x < myvec2_2.x ? 1 : -1,
287  signY = myvec2_1.y < myvec2_2.y ? 1 : -1;
288  vec2int temp = myvec2_1;
289  if (deltaX == 0) {
290  for (int i = 0; i <= deltaY; i++) {
291  setPixel(temp, mych);
292  temp.y += signY;
293  }
294  return;
295  }
296  if (deltaY == 0) {
297  for (int i = 0; i <= deltaX; i++) {
298  setPixel(temp, mych);
299  temp.x += signX;
300  }
301  return;
302  }
303  if (deltaY == deltaX) {
304  for (int i = 0; i <= deltaX; i++) {
305  setPixel(temp, mych);
306  temp.x += signX;
307  temp.y += signY;
308  }
309  return;
310  }
311 
312  int error = deltaX - deltaY;
313  setPixel(myvec2_2, mych);
314  while (temp.x != myvec2_2.x || temp.y != myvec2_2.y)
315  {
316  setPixel(temp, mych);
317  int error2 = error * 2;
318  if (error2 > -deltaY)
319  {
320  error -= deltaY;
321  temp.x += signX;
322  }
323  if (error2 < deltaX)
324  {
325  error += deltaX;
326  temp.y += signY;
327  }
328  }
329 
330 }

380  drawLine(pointpos, prevpos, mych);

390 void AEGraphic::drawCircle(const vec2int& myvec2, const int radius, const CHAR_INFO& mych) {
391 
392 
393  switch (radius)
394  {
395  //just dot
396  case 0:
397  return;
398  break;
399 
400  //crosshair without middle dot
401  case 1:
402  setPixel({ myvec2.x - 1, myvec2.y }, mych);
403  setPixel({ myvec2.x + 1, myvec2.y }, mych);
404  setPixel({ myvec2.x, myvec2.y - 1 }, mych);
405  setPixel({ myvec2.x, myvec2.y + 1 }, mych);
406  return;
407  break;
408 
409  //5x5 square without corner pixels
410  case 2:
411  //top line
412  setPixel({ myvec2.x - 1, myvec2.y - 2 }, mych);
413  setPixel({ myvec2.x, myvec2.y - 2 }, mych);
414  setPixel({ myvec2.x + 1, myvec2.y - 2 }, mych);
415  //bottom line
416  setPixel({ myvec2.x - 1, myvec2.y + 2 }, mych);
417  setPixel({ myvec2.x, myvec2.y + 2 }, mych);
418  setPixel({ myvec2.x + 1, myvec2.y + 2 }, mych);
419  //left line
420  setPixel({ myvec2.x - 2, myvec2.y - 1 }, mych);
421  setPixel({ myvec2.x - 2, myvec2.y}, mych);
422  setPixel({ myvec2.x - 2, myvec2.y + 1 }, mych);
423  //right line
424  setPixel({ myvec2.x + 2, myvec2.y - 1 }, mych);
425  setPixel({ myvec2.x + 2, myvec2.y }, mych);
426  setPixel({ myvec2.x + 2, myvec2.y + 1 }, mych);
427  return;
428  break;
429  default:
430  int x = 0;
431  int y = radius;
432  int delta = 1 - 2 * radius;
433  int error;
434 
435  while (y >= x) {
436  setPixel({ myvec2.x + x, myvec2.y + y }, mych);
437  setPixel({ myvec2.x + x, myvec2.y - y }, mych);
438  setPixel({ myvec2.x - x, myvec2.y + y }, mych);
439  setPixel({ myvec2.x - x, myvec2.y - y }, mych);
440  setPixel({ myvec2.x + y, myvec2.y + x }, mych);
441  setPixel({ myvec2.x + y, myvec2.y - x }, mych);
442  setPixel({ myvec2.x - y, myvec2.y + x }, mych);
443  setPixel({ myvec2.x - y, myvec2.y - x }, mych);
444  error = 2 * (delta + y) - 1;
445  if ((delta < 0) && (error <= 0)) {
446  delta += 2 * ++x + 1;
447  continue;
448  }
449  if ((delta > 0) && (error > 0)) {
450  delta -= 2 * --y + 1;
451  continue;
452  }
453  delta += 2 * (++x - --y);
454  }
455  break;
456  }

516 
517 
518  timeend = std::chrono::system_clock::now();
519  timeelapsed = std::chrono::duration<float>(timeend - timestart).count();
520  if (timeelapsed > 1.0f) {
521  //i -= previ;
522  graph_fps = i - previ;
523  previ = i;
524  timestart = timeend;
525 
526  }
527  m_fr.sleep();
528  i++;
529  }
530  artyk::utils::normal_log(m_logptr, "Closed main drawing thread", LOG_SUCCESS, m_modulename);
531  graph_fps = 0;
