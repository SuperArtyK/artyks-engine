179 inline int __initengine_keyboard() {
180  artyk::app_startstatus = 0;
181  std::memset(AEKeyboard::m_keyOld, 0, 256 * sizeof(short));
182  std::memset(AEKeyboard::m_keys, 0, 256 * sizeof(AEKBKey));
183 
184  for (short i = 0; i < 256; i++) {
185  AEKeyboard::m_keys[i].m_keyid = i;
186  for (short a = 0; a < (sizeof(AEKeyboard::m_keycodes) / sizeof(AEKeyboard::m_keycodes[0])); a++) {
187  if (i == AEKeyboard::m_keycodes[a].second) {
188  AEKeyboard::m_keys[i].m_name = AEKeyboard::m_keycodes[a].first;
189  }
190  }
191  if (AEKeyboard::m_keys[i].m_name == nullptr) {
192  AEKeyboard::m_keys[i].m_name = "---";
193  }
194 }
195  //done with keyboard
196  artyk::app_startstatus = 1;
197  return 0;
198 }
